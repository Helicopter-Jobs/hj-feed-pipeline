SOURCES_FILE = "sources.txt"
OUT_XML = "feed.xml"

OPENAI_API_KEY = os.environ.get("OPENAI_API_KEY", "").strip()
if not OPENAI_API_KEY:
    raise SystemExit(

USER_AGENT = (
    "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 "
    "(KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36"
)

BROWSER_TIMEOUT_MS = 45_000
# With your current scale, keep this conservative so the run finishes reliably.
MAX_LISTING_LINKS_PER_SOURCE = 250
MAX_JOB_PAGES_TOTAL = 600  # safety cap across all sources
# Reduce token/cost:
MAX_TEXT_CHARS_TO_LLM = 35_000

# Your exact required employer names:
EMPLOYER_MAP = {
    "castleair.co.uk": "Castle Air Aviation",
    "jobs.heliservice.de": "HeliService",
    "sloanehelicopters.com": "Sloane Helicopters",
    "nhv-group.jobtoolz.com": "NHV Group",

    "allcareers-quanta.icims.com": "PJ Helicopters",
    "hphelicopters.com": "HIGH PERFORMANCE HELICOPTERS",
}

CATEGORY_ENUM = ["Pilot", "Maintenance", "Medical", "Dispatch", "Operations", "Other"]

ATS_HINTS = [
    "myworkdayjobs.com",
    "icims.com",
    "jobtoolz.com",
    "smartrecruiters.com",
    "greenhouse.io",
    "lever.co",
    "workable.com",
]

JOB_LINK_HINTS = [
    "/job",
    "/jobs",
    "/career",
    "/careers",
    "/position",
    "/positions",
    "/vacancy",
    "/vacancies",
    "jobid=",

    "/en-us/careers/job",
]

PDF_EXT = ".pdf"


# =========
# HELPERS
# =========
def normalize_space(s: str) -> str:
    s = s.replace("\r\n", "\n").replace("\r", "\n")
    s = re.sub(r"[ \t]+", " ", s)
    s = re.sub(r"\n{3,}", "\n\n", s)
    return s.strip()


def domain_from_url(url: str) -> str:
    try:
        host = urlparse(url).netloc.lower()
        host = host.replace("www.", "")
        return host
    except Exception:
        return ""


def employer_for_domain(dom: str) -> str:
    if dom in EMPLOYER_MAP:
        return EMPLOYER_MAP[dom]
    # suffix match to handle subdomains if you add them later
    for k, v in EMPLOYER_MAP.items():
        if dom.endswith(k):
            return v
    return dom or "Unknown"


def read_sources() -> List[str]:
    """
    Reads sources.txt.
    Robust to:
      - one URL per line (ideal)
      - accidental "all in one line" paste (splits on whitespace)
    """
    if not os.path.exists(SOURCES_FILE):
        raise SystemExit("sources.txt not found in repo root.")

    raw = open(SOURCES_FILE, "r", encoding="utf-8").read().strip()
    if not raw:
        raise SystemExit("sources.txt is empty.")

    # Remove comments and common junk
    raw = "\n".join([ln for ln in raw.splitlines() if ln.strip() and not ln.strip().startswith("#")]).strip()

    # If it still looks like one long line, split on whitespace
    if "\n" not in raw and "http" in raw and " " in raw:
        parts = raw.split()
        urls = [p.strip() for p in parts if p.strip().startswith("http")]
        return urls

    urls: List[str] = []
    for line in raw.splitlines():
        u = line.strip()
        if not u:
            continue
        # If someone pasted multiple URLs on one line, split them
        if "http" in u and " " in u:
            for p in u.split():
                if p.strip().startswith("http"):
                    urls.append(p.strip())
        else:
            urls.append(u)

    # De-dupe preserving order
    seen = set()
    out = []
    for u in urls:
        if u not in seen:
            out.append(u)
            seen.add(u)
    return out


def is_likely_job_link(url: str) -> bool:
    u = url.lower()
    if u.startswith("mailto:") or u.startswith("javascript:"):
        return False
    if u.endswith(PDF_EXT):
        return True
    if any(h in u for h in ATS_HINTS):
        return True
    return any(h in u for h in JOB_LINK_HINTS)


def extract_text_from_html(html_content: str) -> str:
    soup = BeautifulSoup(html_content, "lxml")

    # remove noise
    for tag in soup(["script", "style", "noscript", "svg", "canvas"]):
        tag.decompose()

    text = soup.get_text("\n")
    return normalize_space(text)[:MAX_TEXT_CHARS_TO_LLM]


def fetch_pdf_text(url: str) -> str:
    r = requests.get(url, timeout=60, headers={"User-Agent": USER_AGENT})
    r.raise_for_status()
    tmp = "/tmp/job.pdf"
    with open(tmp, "wb") as f:
        f.write(r.content)
    text = pdf_extract_text(tmp) or ""
    return normalize_space(text)[:MAX_TEXT_CHARS_TO_LLM]


def rfc2822_now() -> str:
    # JBoard expiry is your 30 days; pubDate is just for importer sanity
    return datetime.now(timezone.utc).strftime("%a, %d %b %Y %H:%M:%S GMT")


def rss_escape(s: str) -> str:
    return html.escape(s or "", quote=True)


def stable_guid(url: str) -> str:
    # Use URL as stable external identifier
    return url


def dedupe_urls(urls: List[str]) -> List[str]:
    seen = set()
    out = []
    for u in urls:
        if u not in seen:
            out.append(u)
            seen.add(u)
    return out


def dedupe_jobs_by_guid(jobs: List[Dict]) -> List[Dict]:
    seen = set()
    out = []
    for j in jobs:
        g = (j.get("guid") or "").strip().lower()
        if not g or g in seen:
            continue
        seen.add(g)
        out.append(j)
    return out


# ==========================
# OPENAI EXTRACTION (LLM)
# ==========================
def openai_extract_job(source_url: str, raw_text: str, employer: str) -> Dict:
    """
    Uses OpenAI Responses API.
    Returns a dict with:
      title, employer, location, remote, apply_url, category, description, salary_line
    """
    instructions = f"""
Return ONLY valid JSON with these keys:
title (string),
employer (string),
location (string),
remote (boolean),
apply_url (string),
category (one of {CATEGORY_ENUM}),
description (string),
salary_line (string or empty)

Rules:
- Do NOT guess. Use only the provided text.
- employer MUST be exactly: "{employer}"
- location:
  - If an explicit location is stated, use it (city/state/base/region).
  - If location is not specified anywhere, use the country if clearly stated; otherwise "Not specified".
- remote: true only if the posting explicitly says Remote/Hybrid/Telecommute; else false.
- apply_url: use a clear apply link if present; otherwise use source_url.
- salary_line:
  - Only if pay is explicitly stated. Format exactly like:
    Pay: $X–$Y/hr (from posting)
    Pay: $X–$Y/yr (from posting)
  - Otherwise return "".
- description: clean plain text, keep bullets; no HTML.
"""

    payload = {
        "model": "gpt-4o-mini",
        "input": [
            {"role": "system", "content": "Extract structured job fields for a helicopter job board. Be precise; do not guess."},
            {"role": "user", "content": f"source_url: {source_url}\n\n{instructions}\n\nJOB PAGE TEXT:\n{raw_text}"},
        ],
        "temperature": 0.0,
    }

    def call(model: str, timeout_s: int) -> Dict:
        payload["model"] = model
        resp = requests.post(
            "https://api.openai.com/v1/responses",
            headers={
                "Authorization": f"Bearer {OPENAI_API_KEY}",
                "Content-Type": "application/json",
            },
            data=json.dumps(payload),
            timeout=timeout_s,
        )
        resp.raise_for_status()
        data = resp.json()
        text_out = ""
        for item in data.get("output", []):
            for c in item.get("content", []):
                if c.get("type") == "output_text":
                    text_out += c.get("text", "")
        text_out = text_out.strip()
        return json.loads(text_out)

    try:
        job = call("gpt-4o-mini", 90)
    except Exception:
        # fallback once
        job = call("gpt-4o", 120)

    # hard enforce + sanitize
    job["employer"] = employer
    job["source_url"] = source_url
    job["guid"] = stable_guid(source_url)

    if job.get("category") not in CATEGORY_ENUM:
        job["category"] = "Other"

    if not isinstance(job.get("remote"), bool):
        job["remote"] = False

    for k in ["title", "location", "apply_url", "description", "salary_line"]:
        job[k] = str(job.get(k, "")).strip()

    if not job["apply_url"]:
        job["apply_url"] = source_url

    # Salary line: keep it short and single-line
    if job.get("salary_line"):
        job["salary_line"] = normalize_space(job["salary_line"]).replace("\n", " ").strip()

    return job


# ==========================
# RSS XML GENERATION
# ==========================
def build_rss_feed(jobs: List[Dict]) -> str:
    pubdate = rfc2822_now()

    out: List[str] = []
    out.append('<?xml version="1.0" encoding="UTF-8"?>')
    out.append('<rss version="2.0">')
    out.append("  <channel>")
    out.append("    <title>Helicopter-Jobs Aggregated Feed</title>")
    out.append("    <link>https://helicopter-jobs.com</link>")
    out.append("    <description>Direct-employer helicopter jobs</description>")

    for j in jobs:
        title = rss_escape(j.get("title", ""))
        employer = rss_escape(j.get("employer", ""))
        link = rss_escape(j.get("apply_url", j.get("source_url", "")))
        guid = rss_escape(j.get("guid", j.get("source_url", "")))
        category = rss_escape(j.get("category", "Other"))
        location = rss_escape(j.get("location", "Not specified") or "Not specified")
        remote = "true" if j.get("remote") else "false"

        desc = j.get("description", "").strip()
        if j.get("salary_line"):
            desc = f"{j['salary_line']}\n\n{desc}".strip()

        out.append("    <item>")
        out.append(f"      <title>{title}</title>")
        out.append(f"      <employer>{employer}</employer>")
        out.append(f"      <link>{link}</link>")
        out.append(f'      <guid isPermaLink="true">{guid}</guid>')
        out.append(f"      <pubDate>{pubdate}</pubDate>")
        out.append(f"      <category>{category}</category>")
        out.append(f"      <location>{location}</location>")
        out.append(f"      <remote>{remote}</remote>")
        out.append("      <description><![CDATA[")
        out.append(desc)
        out.append("]]></description>")
        out.append("    </item>")

    out.append("  </channel>")
    out.append("</rss>")
    return "\n".join(out)


# ==========================
# CRAWLING (PLAYWRIGHT)
# ==========================
def collect_links_from_listing(base_url: str, listing_html: str) -> List[str]:
    soup = BeautifulSoup(listing_html, "lxml")
    links: List[str] = []
    for a in soup.find_all("a", href=True):
        href = (a.get("href") or "").strip()
        if not href:
            continue
        full = urljoin(base_url, href)
        if is_likely_job_link(full):
            links.append(full)
    links = dedupe_urls(links)
    return links[:MAX_LISTING_LINKS_PER_SOURCE]


def main():
    sources = read_sources()
    print(f"Loaded {len(sources)} sources")

    jobs: List[Dict] = []
    seen_job_urls: Set[str] = set()

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        context = browser.new_context(user_agent=USER_AGENT)
        page = context.new_page()

        for src in sources:
            dom = domain_from_url(src)
            employer = employer_for_domain(dom)

            print(f"\nSOURCE: {src}")
            print(f"Employer: {employer}")

            try:
                page.goto(src, timeout=BROWSER_TIMEOUT_MS, wait_until="networkidle")
                time.sleep(1.0)
                listing_html = page.content()
            except Exception as e:
                print(f"  !! Could not load listing page: {e}")
                continue

            candidate_links = collect_links_from_listing(src, listing_html)
            print(f"  Found {len(candidate_links)} candidate links")

            # If we found nothing, still try to treat source page as a single job page (rare)
            if not candidate_links:
                candidate_links = [src]

            for job_url in candidate_links:
                if len(seen_job_urls) >= MAX_JOB_PAGES_TOTAL:
                    print("Reached MAX_JOB_PAGES_TOTAL cap; stopping crawl.")
                    break

                if job_url in seen_job_urls:
                    continue
                seen_job_urls.add(job_url)

                try:
                    if job_url.lower().endswith(PDF_EXT):
                        raw_text = fetch_pdf_text(job_url)
                        source_url = job_url
                    else:
                        page.goto(job_url, timeout=BROWSER_TIMEOUT_MS, wait_until="networkidle")
                        time.sleep(0.8)
                        raw_text = extract_text_from_html(page.content())
                        source_url = job_url

                    if len(raw_text) < 200:
                        continue

                    job = openai_extract_job(source_url=source_url, raw_text=raw_text, employer=employer)
                    jobs.append(job)
                    print(f"   + {job.get('title','(no title)')[:90]}")
                except Exception as e:
                    print(f"   - Failed job page: {job_url} | {str(e)[:180]}")
                    continue

        browser.close()

    jobs = dedupe_jobs_by_guid(jobs)
    print(f"\nExtracted {len(jobs)} unique jobs")

    rss = build_rss_feed(jobs)
    with open(OUT_XML, "w", encoding="utf-8") as f:
        f.write(rss)

    print(f"Wrote {OUT_XML}")


if __name__ == "__main__":
    main()
